function handleVideo(call) { // Check if we already have a video for this peer to avoid duplicates if (document.getElementById(`video-${call.peer}`)) return; const v = document.createElement('video'); v.id = `video-${call.peer}`; // Give it a unique ID v.autoplay = true; v.playsInline = true; call.on('stream', remoteStream => { v.srcObject = remoteStream; mainStage.appendChild(v); // Store the connection if (!connections[call.peer]) connections[call.peer] = {}; connections[call.peer].call = call; }); call.on('close', () => { v.remove(); delete connections[call.peer]; }); }
